Set Up CON on macOS

# Install dependencies
brew install yosys ghdl gtkwave icarus-verilog

# Get sv2v
git clone https://github.com/zachjs/sv2v.git
cd sv2v
make

# Check installation
sv2v --version
yosys --version

brew install icarus-verilog

Using GTKWave on macOS

git clone git@github.com:gtkwave/gtkwave.git
cd gtkwave
    
brew install desktop-file-utils shared-mime-info \
gobject-introspection gtk-mac-integration \
meson ninja pkg-config gtk+3 gtk4
    
mkdir build
cd build
    
meson setup . ..
    
ninja
    
sudo ninja install
    
gtkwave
    
# Clean up
ninja clean
cd ..
rm -rf build
